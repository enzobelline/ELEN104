problem 1c

z = linspace(0.00001,100,1100);
z1 = linspace(1,100,1100);
Ez = (ps/(2*e))*(1-(z./sqrt(a.^2+z.^2)));
Ep = (ps*pi*a^2)./(4*e*pi*z1.^2);

figure(1)
hold on
loglog(z,Ez,'black')
loglog(z,Ep,'g')
legend('Magnitude of Ez','Infinitive Plate Line','Point Charge Line')
xlim([0.01, 100])
ylim([10^9, 10^12])
hold off
